Cut the rope
// Time : O(3^n) int maxCuts(int n, int a, int b, int c) { if(n == 0) return 0; if(n <= -1) return -1; int res = max(maxCuts(n-a, a, b, c), max(maxCuts(n-b, a, b, c), maxCuts(n-c, a, b, c))); if(res == -1) return -1; return res + 1; }

// Time : 2 ^ n void countSubset(int arr, int n , int sum){ if(n==0) return sum == 0? 1:0; return countSubset(arr,n-1,sum) + countSubset(arr,n-1,sum - arr[n-1]); }

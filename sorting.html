Insertion Sort (Stable, Inplace)
void insertionSort(int arr[], int n){ for(int i=0; i<n; i++){ int key = arr[i]; int j = i - 1; while(j >= 0 && arr[j] > key){ // if you put arr[j] `>=` key this would make it unstable arr[j+1] = arr[j]; j--; } arr[j+1] = key; } }

Merge Sort (Stable)
void merge(int arr[], int l, int m, int h){ int n1 = m-l+1, n2 = h-m; // get the size of left and right subarray int left[n1], right[n2]; for(int i=0; i<n1; i++) left[i] = arr[i+l]; for(int i=0; i<n2; i++) right[i] = arr[m+1+j]; int i=0; j=0; k=l; while(i<n1 && j<n2){ if(left[i] <= right[j]) arr[k++] = left[i++]; else arr[k++] = right[j++]; } while(i<n1) arr[k++] = left[i++]; while(j<n2) arr[k++] = right[j++]; } void mergeSort(int arr[], int l, int h){ while(l<h){ int m = l+(h-l)/2; mergeSort(arr, l, m); // from left to mid mergeSort(arr, m+1, h); // from mid + 1 to right merge(arr, l, m, h); } }

Count inversion in array
Count inversion means: A pair (a[i], b[j]) forms an inversion when i<j & a[i]>b[j].
int countAndMerge(int arr[], int l, int m, int r) { int n1=m-l+1, n2=r-m; int left[n1],right[n2]; for(int i=0;i<n1;i++) left[i]=arr[i+l]; for(int j=0;j<n2;j++) right[j]=arr[m+1+j]; int res=0,i=0,j=0,k=l; while(i<n1 && j<n2){ if(left[i]<=right[j]) {arr[k++]=left[i++];} else{ arr[k++]=right[j++]; res=res+(n1-i); } } while(i<n1) arr[k++]=left[i++]; while(j<n2) arr[k++]=right[j++]; return res; } int countInv(int arr[], int l, int r) { int res = 0; if (l<r) { int m = (r + l) / 2; res += countInv(arr, l, m); res += countInv(arr, m + 1, r); res += countAndMerge(arr, l, m , r); } return res; }

Minimum Platform
Given arrival and departure times of all trains that reach a railway station. Find the minimum number of platforms required for the railway station so that no train is kept waiting. Consider that all the trains arrive on the same day and leave on the same day. Arrival and departure time can never be the same for a train but we can have arrival time of one train equal to departure time of the other. At any given instance of time, same platform can not be used for both departure of a train and arrival of another train. In such cases, we need different platforms.
int findPlatform(int arr[], int dep[], int n){ // Sort arrival and departure arrays sort(arr, arr+n); sort(dep, dep+n); int i=1, j=0, res=1, plat_need=1; while(i<n && j<n){ if(arr[i]<=dep[j]){ i++; plat_need++; if(plat_need>res) res = plat_need; } else{ j++; plat_need--; } } return res; }
